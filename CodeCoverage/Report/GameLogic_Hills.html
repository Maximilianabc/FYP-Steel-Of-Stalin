 554        public IEnumerable<Tile> GetPath(Tile start, Tile end, PathfindingOptimization opt = PathfindingOptimization.LEA
 555        {
 556            if (this is Personnel && opt == PathfindingOptimization.LEAST_FUEL_COST)
 557            {
 558                this.LogError("Only units with fuel capacity can be used with fuel cost optimization for pathfinding.");
 559                return new List<Tile>();
 560            }
 561
 562            bool is_aerial = this is Aerial;
 563            WeightedTile w_start = start.ConvertToWeightedTile(Consumption, opt, end, is_aerial);
 564            WeightedTile w_end = end.ConvertToWeightedTile(Consumption, opt, end, is_aerial);
 565
 566            List<WeightedTile> active = new List<WeightedTile>();
 567            List<WeightedTile> visited = new List<WeightedTile>();
 568            List<Tile> path = new List<Tile>();
 569            Func<WeightedTile, decimal> sort =
 570                opt == PathfindingOptimization.LEAST_SUPPLIES_COST
 571                ? new Func<WeightedTile, decimal>(w => w.SuppliesCostDistance)
 572                : w => w.FuelCostDistance;
 573
 574            active.Add(w_start);
 575            while (active.Any())
 576            {
 577                WeightedTile w_check = active.OrderBy(sort).FirstOrDefault();
 578                if (w_check == w_end
 579                    || w_check.SuppliesCostSoFar > Carrying.Supplies
 580                    || (Consumption.Fuel > 0 && w_check.FuelCostSoFar > Carrying.Fuel)
 581                    || w_check.DistanceSoFar > Maneuverability.Speed)
 582                {
 583                    while (w_check.Parent != null)
 584                    {
 585                        path.Add(Map.Instance.GetTile(w_check.CubeCoOrds));
 586                        w_check = w_check.Parent;
 587                    }
 588                    return path;
 589                }
 590                visited.Add(w_check);
 591                _ = active.Remove(w_check);
 592
 593                List<WeightedTile> w_neighbours = new List<WeightedTile>();
 594                GetAccessibleNeigbours(w_check.CubeCoOrds).ToList().ForEach(t => w_neighbours.Add(t.ConvertToWeightedTil
 595
 596                w_neighbours.ForEach(n =>
 597                {
 598                    if (!visited.Where(v => v == n).Any())
 599                    {
 600                        WeightedTile w_exist = active.Where(a => a == n).FirstOrDefault();
 601                        if (w_exist != null)
 602                        {
 603                            // exist in active list
 604                            bool moreExpensive =
 605                                opt == PathfindingOptimization.LEAST_SUPPLIES_COST
 606                                ? w_exist.SuppliesCostDistance > w_check.SuppliesCostDistance
 607                                : w_exist.FuelCostDistance > w_check.FuelCostDistance;
 608

 610                            if (moreExpensive)
 611                            {
 612                                _ = active.Remove(w_exist);
 613                            }
 614                        }
 615                        else
 616                        {
 617                            // add the neigbour to active list
 618                            active.Add(n);
 619                        }
 620                    }
 621                });
 622            }
